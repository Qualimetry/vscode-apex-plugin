Compliant Solution

public class LegacyService {
    public void process() {
        // Fix the underlying issues instead of suppressing
        Logger.info('Processing started');
        String q = 'SELECT Id FROM Account WHERE Name = \''
            + String.escapeSingleQuotes(input) + '\'';
    }
}
